# 冒泡

```c
int main() {
  int a[10] = {2, 3, 512, 12, 65, 1, 8, 9, 50, 2};
  int l = 10;

  for (int i = 0; i < l; i++) {
    for (int j = 0; j < l - 1; j++) {
      if (a[j] > a[j + 1]) {
        int t = a[j];
        a[j] = a[j + 1];
        a[j + 1] = t;
      }
    }
  }

  for (int i = 0; i < l; i++) {
    printf("%d ", a[i]);
  }
}
```

# 选择

```c
int main() {
  int a[10] = {2, 3, 512, 12, 65, 1, 8, 9, 50, 2};
  int l = 10;

  for (int i = 0; i < l - 1; i++) {
    int min = i + 1;
    for (int j = i + 1; j < l; j++) {
      if (a[min] > a[j]) {
        min = j;
      }
    }
    if (a[i] > a[min]) {
      int t = a[min];
      a[min] = a[i];
      a[i] = t;
    }
  }

  for (int i = 0; i < l; i++) {
    printf("%d ", a[i]);
  }
}
```
